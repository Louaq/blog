def get_weight_size(path):

def warmup_model(model, device, example_inputs, iterations=200):
    """æ¨¡å‹é¢„çƒ­ï¼Œå‡†å¤‡è¿›è¡Œé«˜æ•ˆæ¨ç†ã€‚"""

    """æµ‹è¯•æ¨¡å‹çš„æ¨ç†å»¶è¿Ÿã€‚"""

def main(opt):
